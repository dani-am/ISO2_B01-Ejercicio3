21  		
22  		//caso 1: MENOR de edad, ESTUDIANDO y NO INDEPENDIZADO
23  		if(cl.getEdad()<18 && cl.isEstudia() && !cl.isIndependizado())
24  			return "Cuenta confort";
25  		
26  		//caso 2: MENOR DE 25, ESTUDIANDO e INDEPENDIZADO

28  			return "Cuenta vamos que tú puedes";
29  		
30  		//Caso 3; MAYOR de edad, TRABAJANDO, NO INDEPENDIZADO
31  		if(cl.getEdad()>18 && !cl.isEstudia())
32  			if(!cl.isIndependizado())
33  				return "Cuenta ahorra ahora que puedes";
34  			else
35  				return "Cuenta saltando del nido";
36

38  		if(cl.getEdad()>25 && !cl.isEstudia() && !cl.isIndependizado())
39  			return "Cuenta independízate que va siendo hora";
40  		
41  		//Caso 5: MAYOR de 25, TRABAJANDO, INDEPENDIZADO
42  		if(cl.getEdad()>25 && !cl.isEstudia() && cl.isIndependizado())
43  			return "Cuenta bienvenido a la vida adulta";
44
